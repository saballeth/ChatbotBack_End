source .venv/bin/activate          # Windows: .\.venv\Scripts\activate

# 2) Install deps
pip install -r requirements.txt

# 3) Run Ollama in another terminal
ollama serve   # <-- make sure a small model like 'mistral' is pulled

# 4) Start FastAPI
uvicorn main:app --reload
